/**
  * @brief  Provides a delay in microseconds using TIM2.
  * @param  us: Delay in microseconds.
  * @retval None
  */
void delay_us(uint16_t us)
{
    __HAL_TIM_SET_COUNTER(&htim2, 0);
    while (__HAL_TIM_GET_COUNTER(&htim2) < us);
}

// Redirect printf to USB CDC

{
    CDC_Transmit_FS((uint8_t*)ptr, len);
    return len;
}

/**
  * @brief  The application entry point.
  * @retval int
  */
int main(void)
{
  /* USER CODE BEGIN 1 */
  uint32_t distance_cm = 0;
  uint32_t echo_time_us = 0;
  /* USER CODE END 1 */

  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  MX_USB_DEVICE_Init();
  MX_USART2_UART_Init();
  MX_TIM2_Init();
  /* USER CODE BEGIN 2 */
  HAL_TIM_Base_Start(&htim2);

  printf("HC-SR04 Sensor Test Program Started.\r\n");
  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  while (1)
  {
    /* USER CODE END WHILE */

    /* USER CODE BEGIN 3 */
    // 1. Send 10us trigger pulse to HC-SR04 on PA0
    HAL_GPIO_WritePin(GPIOA, GPIO_PIN_0, GPIO_PIN_RESET);
    delay_us(2);
    HAL_GPIO_WritePin(GPIOA, GPIO_PIN_0, GPIO_PIN_SET);
    delay_us(10);
    HAL_GPIO_WritePin(GPIOA, GPIO_PIN_0, GPIO_PIN_RESET);

    // 2. Wait for ECHO pin (PA1) to go HIGH (with timeout)
    uint32_t start_tick = HAL_GetTick();
    echo_time_us = 0; // Reset time
    while(HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_1) == GPIO_PIN_RESET)
    {
        if (HAL_GetTick() - start_tick > 100) // 100ms timeout
        {
            goto measurement_done; // Skip measurement if no echo starts
        }
    }

    // 3. Measure the time the ECHO pin stays HIGH
	__HAL_TIM_SET_COUNTER(&htim2, 0);
	while(HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_1) == GPIO_PIN_SET)
    {
        // Timeout for measurement loop to prevent getting stuck if echo never ends
        if (__HAL_TIM_GET_COUNTER(&htim2) > 23200) // Max range timeout (~4m)
        {
            break;
        }
    }
	echo_time_us = __HAL_TIM_GET_COUNTER(&htim2);


measurement_done:
    // 4. Calculate distance in cm
    // distance (cm) = (echo_time_us / 58)
    distance_cm = echo_time_us / 58;

    // Print distance to USB CDC for debugging
    printf("Echo: %lu us, Distance: %lu cm\r\n", echo_time_us, distance_cm);

    // 5. Control LED on PA7 for debugging
    // PA7 is sink mode, so LOW = ON.
    if(distance_cm > 0 && distance_cm < 20) {
      HAL_GPIO_WritePin(GPIOA, GPIO_PIN_7, GPIO_PIN_RESET); // LED ON
    } else {
      HAL_GPIO_WritePin(GPIOA, GPIO_PIN_7, GPIO_PIN_SET);   // LED OFF
    }

    HAL_Delay(100); // 100ms delay between measurements
  }
  /* USER CODE END 3 */
}
